  <script src="https://cdn.jsdelivr.net/npm/phaser@3/dist/phaser.min.js"></script>

<body>

  <script>

    // ----------------------------------------------------

    // Game configuration

    // ----------------------------------------------------

    const config = {

      type: Phaser.AUTO,

      width: 800,

      height: 600,

      backgroundColor: '#000000',

      physics: {

        default: 'arcade',

        arcade: {

          debug: false,

          // gravity is zero by default

        }

      },

      scene: {

        preload: preload,

        create: create,

        update: update

      }

    };

 

    // Initialize the Phaser Game

    const game = new Phaser.Game(config);

 

    // ----------------------------------------------------

    // Global variables

    // ----------------------------------------------------

    let leftPaddle, rightPaddle, ball;

    let cursors;

    let wKey, sKey;

    let leftScore = 0, rightScore = 0;

    let scoreText;

    let winningScore = 10;

    let gameOver = false;

    let winText;

 

    // ----------------------------------------------------

    // preload()

    //   (No external assets are needed; we'll use simple graphics)

    // ----------------------------------------------------

    function preload() {

      // Nothing to load (we draw shapes at runtime)

    }

 

    // ----------------------------------------------------

    // create()

    //   Set up paddles, ball, controls, and text

    // ----------------------------------------------------

    function create() {

      // Create left paddle (rectangle)

      leftPaddle = this.physics.add.staticImage(50, config.height / 2, null)

        .setOrigin(0.5, 0.5)

        .setDisplaySize(20, 100)

        .refreshBody();

      leftPaddle.body.setSize(20, 100);

 

      // Draw left paddle as white:

      leftPaddle.setFillStyle(0xffffff);

 

      // Create right paddle (rectangle)

      rightPaddle = this.physics.add.staticImage(config.width - 50, config.height / 2, null)

        .setOrigin(0.5, 0.5)

        .setDisplaySize(20, 100)

        .refreshBody();

      rightPaddle.body.setSize(20, 100);

 

      // Draw right paddle as white:

      rightPaddle.setFillStyle(0xffffff);

 

      // Create the ball (circle)

      ball = this.physics.add.image(config.width / 2, config.height / 2, null)

        .setDisplaySize(16, 16)

        .setCollideWorldBounds(true)

        .setBounce(1, 1);

      ball.body.setCircle(8);

      ball.setFillStyle(0xffffff);

 

      // Launch ball in a random direction

      resetBall.call(this);

 

      // Enable collision between ball and paddles

      this.physics.add.collider(ball, leftPaddle, hitPaddle, null, this);

      this.physics.add.collider(ball, rightPaddle, hitPaddle, null, this);

 

      // Set world bounds so the ball can bounce off top/bottom only

      this.physics.world.setBoundsCollision(true, true, true, true);

 

      // Input

      cursors = this.input.keyboard.createCursorKeys();

      wKey = this.input.keyboard.addKey(Phaser.Input.Keyboard.KeyCodes.W);

      sKey = this.input.keyboard.addKey(Phaser.Input.Keyboard.KeyCodes.S);

 

      // Score text

      scoreText = this.add.text(config.width / 2, 20, '0 : 0', {

        fontSize: '32px',

        fill: '#ffffff'

      }).setOrigin(0.5, 0);

 

      // Win text (hidden initially)

      winText = this.add.text(config.width / 2, config.height / 2, '', {

        fontSize: '48px',

        fill: '#ffff00'

      }).setOrigin(0.5).setVisible(false);

    }

 

    // ----------------------------------------------------

    // update()

    //   Runs every frame: move paddles, check scores, etc.

    // ----------------------------------------------------

    function update() {

      if (gameOver) {

        // If game is over, stop updating

        return;

      }

 

      // Left paddle movement (W/S)

      if (wKey.isDown && leftPaddle.y > leftPaddle.displayHeight / 2) {

        leftPaddle.y -= 5;

        leftPaddle.body.updateFromGameObject();

      } else if (sKey.isDown && leftPaddle.y < config.height - leftPaddle.displayHeight / 2) {

        leftPaddle.y += 5;

        leftPaddle.body.updateFromGameObject();

      }

 

      // Right paddle movement (Up/Down)

      if (cursors.up.isDown && rightPaddle.y > rightPaddle.displayHeight / 2) {

        rightPaddle.y -= 5;

        rightPaddle.body.updateFromGameObject();

      } else if (cursors.down.isDown && rightPaddle.y < config.height - rightPaddle.displayHeight / 2) {

        rightPaddle.y += 5;

        rightPaddle.body.updateFromGameObject();

      }

 

      // Check left/right bounds of the ball for scoring

      if (ball.x < 0) {

        // Right player scores

        rightScore += 1;

        updateScore.call(this);

      } else if (ball.x > config.width) {

        // Left player scores

        leftScore += 1;

        updateScore.call(this);

      }

    }

 

    // ----------------------------------------------------

    // resetBall()

    //   Put ball in center and launch in a random direction

    // ----------------------------------------------------

    function resetBall() {

      ball.setPosition(config.width / 2, config.height / 2);

      // Randomize initial velocity: X either left or right, Y small random

      const speed = 250;

      const angle = Phaser.Math.Between(-30, 30);

      const dirX = Phaser.Math.Between(0, 1) === 0 ? -1 : 1;

      this.physics.velocityFromAngle(angle, speed, ball.body.velocity);

      ball.body.velocity.x = dirX * Math.abs(ball.body.velocity.x);

    }

 

    // ----------------------------------------------------

    // hitPaddle(ball, paddle)

    //   Called when ball collides with a paddle: slightly adjust angle

    // ----------------------------------------------------

    function hitPaddle(ball, paddle) {

      // Increase ball speed slightly on each paddle hit (max cap)

      const maxSpeed = 600;

      let newVelX = ball.body.velocity.x * 1.05;

      let newVelY = ball.body.velocity.y * 1.05;

      if (Math.abs(newVelX) > maxSpeed) {

        newVelX = maxSpeed * Phaser.Math.Sign(newVelX);

      }

      if (Math.abs(newVelY) > maxSpeed) {

        newVelY = maxSpeed * Phaser.Math.Sign(newVelY);

      }

      ball.body.setVelocity(newVelX, newVelY);

    }

 

    // ----------------------------------------------------

    // updateScore()

    //   Update score display, check for win, and reset ball

    // ----------------------------------------------------

    function updateScore() {

      scoreText.setText(`${leftScore} : ${rightScore}`);

 

      // Check for win condition

      if (leftScore >= winningScore || rightScore >= winningScore) {

        gameOver = true;

        const winner = leftScore > rightScore ? 'Left Player Wins!' : 'Right Player Wins!';

        winText.setText(winner + '\nPress SPACE to Restart').setVisible(true);

 

        // Stop the ball

        ball.body.setVelocity(0, 0);

 

        // Listen for SPACE to restart

        this.input.keyboard.once('keydown-SPACE', () => {

          leftScore = 0;

          rightScore = 0;

          scoreText.setText('0 : 0');

          winText.setVisible(false);

          gameOver = false;

          resetBall.call(this);

        });

      } else {

        // No winner yet: reset ball and continue

        resetBall.call(this);

      }

    }

  </script>
